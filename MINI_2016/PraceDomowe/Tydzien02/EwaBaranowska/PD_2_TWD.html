EPSG <- make_EPSG()
EPSG[grepl("WGS 84$", EPSG$note), ]
##      code     note                                         prj4
## 249  4326 # WGS 84          +proj=longlat +datum=WGS84 +no_defs
## 4890 4978 # WGS 84 +proj=geocent +datum=WGS84 +units=m +no_defs
poland.map <- spTransform(poland.map, CRS("+init=epsg:4326"))

poland.map.gg <- fortify(poland.map)
